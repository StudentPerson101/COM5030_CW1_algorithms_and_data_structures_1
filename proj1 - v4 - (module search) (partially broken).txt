        {

            {"Computer Science with Artificial Intelligence", 1, &ModuleSelect0.ARIN_y1},
            {"Computer Science with Artificial Intelligence", 2, &ModuleSelect0.ARIN_y2},
            {"Computer Science with Artificial Intelligence", 3, &ModuleSelect0.ARIN_y3},
            {"Computer Science", 1, &ModuleSelect0.CS01_y1},
            {"Computer Science", 2, &ModuleSelect0.CS01_y2},
            {"Computer Science", 3, &ModuleSelect0.CS01_y3},

int moduleFunc()
{
    CompModules0 cmSelect0;

        [](unsigned char c) { return std::tolower(c); });
        if (Ans3 == "yes")
        {

            {

        }
        else if (Ans3 == "exit")
        {
            return 0;
        }
        
    }
}
